41 public MLDecisionEngine(MLConfigSO config)
42 {
43 if (config.modelAssets == null || config.modelAssets.Length == 0)
44 {
45 Debug.LogError("[DDA] MLConfigSO: assegnare almeno un modello ONNX in modelAssets!");
46 }

48 // Ordina i modelli per nome (es. dda_multi_Difficulty, then _Experience, ...)
49 var sortedAssets = config.modelAssets.OrderBy(a => a.name).ToArray();
50
51 _env = OrtEnv.Instance();
52 // Crea una sessione per ciascun modello ONNX
53 _sessions = sortedAssets
54 .Select(asset => new InferenceSession(Convert.FromBase64String(asset.text)))
55 .ToArray();
56
57 // Estrae il nome dell'unico input tensor e la dimensione delle feature
58 _inputName = _sessions[0].InputMetadata.Keys.First();
59 var inputDims = _sessions[0].InputMetadata[_inputName].Dimensions;
60 if (inputDims.Length != 2)
61 {
62 Debug.LogError($"[DDA] Input ONNX inaspettato: attesi 2 assi (batch,features), trovati {inputDims.Length}");
63 }
64 _featureCount = inputDims[1];
65
66 // Legge automaticamente i nomi degli output (graph.output) da ciascun modello
67 _metricNames = _sessions.Select(s => s.OutputMetadata.Keys.First()).ToArray();
68 if (_metricNames.Length != _featureCount)
69 {
70 Debug.LogWarning($"[DDA] Numero modelli ({_metricNames.Length}) diverso da featureCount ({_featureCount})");
71 }

85 public Dictionary<string, float> Evaluate(Dictionary<string, float> normalizedMetrics)
86 {
87 // Popola il buffer secondo l'ordine di _metricNames
88 var buffer = new float[_featureCount];
89 for (int i = 0; i < _featureCount; i++)
90 {
91 var key = _metricNames[i];
92 buffer[i] = normalizedMetrics.TryGetValue(key, out var v) ? v : 0f;
93 }
94
95 // Crea un DenseTensor<float> di shape [1, featureCount]
96 var inputTensor = new DenseTensor<float>(buffer, new[] { 1, _featureCount });
97 var container = NamedOnnxValue.CreateFromTensor(_inputName, inputTensor);
98
99 var result = new Dictionary<string, float>(_sessions.Length);
100 // Esegue inferenza su ciascun modello e mappa l'output al nome metrica
101 for (int i = 0; i < _sessions.Length; i++)
102 {
103 using var outputs = _sessions[i].Run(new[] { container });
104 var outTensor = outputs.First().AsTensor<float>();
105 result[_metricNames[i]] = outTensor.Length > 0 ? outTensor.GetValue(0) : 0f;
106 }
107 return result;
108 }
