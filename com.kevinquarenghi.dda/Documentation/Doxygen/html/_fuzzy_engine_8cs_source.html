40 public Dictionary<string, float> Infer(Dictionary<string, float> inputs)
41 {
42 // 1. Fuzzificazione
43 var fuzzified = new Dictionary<(string var, string term), float>();
44 foreach (var variable in _config.variables)
45 {
46 if (!inputs.TryGetValue(variable.name, out var x))
47 continue;
48 foreach (var term in variable.terms)
49 fuzzified[(variable.name, term.label)] = FuzzifyTerm(x, term.points);
50 }
51
52 // 2. Valutazione regole con tracking delle variabili attivate
53 var actionTermActivations = new Dictionary<(string var, string term), List<float>>();
54 for (int i = 0; i < _config.rules.Count; i++)
55 {
56 var rule = _config.rules[i];
57 float activation = 0f;
58 bool first = true;
59
60 // AND/OR sequenziali tra condizioni
61 foreach (var cond in rule.conditions)
62 {
63 fuzzified.TryGetValue((cond.variable, cond.term), out var μ);
64 if (first)
65 {
66 activation = μ;
67 first = false;
68 }
69 else if (cond.useOrWithPrev)
70 activation = Mathf.Max(activation, μ);
71 else
72 activation = Mathf.Min(activation, μ);
73 }
74
75 // Se la regola ha attivazione positiva, invia evento e accumula le azioni
76 if (activation > 0f)
77 {
78 DecisionEngineEvents.NotifyRuleActivated(i, activation);
79
80 foreach (var act in rule.actions)
81 {
82 var key = (act.variable, act.term);
83 if (!actionTermActivations.TryGetValue(key, out var list))
84 {
85 list = new List<float>();
86 actionTermActivations[key] = list;
87 }
88 list.Add(activation);
89 }
90 }
91 }
92
93 // 3. Costruzione degli output fuzzy solo per termini con α > 0
94 var fuzzyTermOutput = new Dictionary<(string var, string term), float>();
95 foreach (var kv in actionTermActivations)
96 {
97 float alphaMax = kv.Value.Max();
98 if (alphaMax > 0f)
99 fuzzyTermOutput[kv.Key] = alphaMax;
100 }
101
102 // 4. Defuzzificazione solo delle variabili effettivamente attivate
103 var crispOutput = new Dictionary<string, float>();
104 var triggeredVars = fuzzyTermOutput
105 .Keys
106 .Select(k => k.var)
107 .Distinct();
108
109 foreach (var varName in triggeredVars)
110 {
111 var variableConfig = _config.variables
112 .First(v => v.name == varName);
113
114 // raccolgo le α dei termini attivati
115 var termAlphas = fuzzyTermOutput
116 .Where(kv => kv.Key.var == varName)
117 .ToDictionary(kv => kv.Key.term, kv => kv.Value);
118
119 float defuzzed = Defuzzify(variableConfig.terms, termAlphas);
120 crispOutput[varName] = defuzzed;
121 }

137 private float FuzzifyTerm(float x, float[] p)
138 {
139 // triangolo [a,b,c] o trapezio [a,b,c,d]
140 if (p.Length == 3)
141 {
142 var a = p[0];
143 var b = p[1];
144 var c = p[2];
145
146 // Shoulder sinistro (a==b): plateau completo fino a b
147 if (Mathf.Approximately(a, b))
148 {
149 if (x <= b) return 1f; // tutto sotto b ha membership 1
150 if (x >= c) return 0f; // tutto sopra c è 0
151 return (c - x) / (c - b); // decresce linearmente da b→c
152 }
153 // Shoulder destro (b==c): plateau completo da b in poi
154 if (Mathf.Approximately(b, c))
155 {
156 if (x >= b) return 1f; // tutto sopra b ha membership 1
157 if (x <= a) return 0f; // tutto sotto a è 0
158 return (x - a) / (b - a); // cresce linearmente da a→b
159 }
160 // Triangolo “normale”
161 if (x <= a || x >= c) return 0f;
162 if (Mathf.Approximately(x, b)) return 1f;
163 return x < b
164 ? (x - a) / (b - a)
165 : (c - x) / (c - b);
166 }
167
168 if (p.Length == 4)
169 {
170 var a = p[0];
171 var b = p[1];
172 var c = p[2];
173 var d = p[3];
174
175 // Trapezio con shoulder sinistro
176 if (Mathf.Approximately(a, b))
177 {
178 if (x <= b) return 1f;
179 if (x >= d) return 0f;
180 if (x <= c) return 1f; // plateau piatto tra b e c
181 return (d - x) / (d - c);
182 }
183 // Trapezio con shoulder destro
184 if (Mathf.Approximately(c, d))
185 {
186 if (x >= c) return 1f;
187 if (x <= a) return 0f;
188 if (x >= b) return 1f; // plateau piatto tra b e c
189 return (x - a) / (b - a);
190 }
191 // Trapezio “normale”
192 if (x <= a || x >= d) return 0f;
193 if (x >= b && x <= c) return 1f;
194 return x < b
195 ? (x - a) / (b - a)
196 : (d - x) / (d - c);
197 }
198
199 return 0f;
200 }

214 private float Defuzzify(List<FuzzyTerm> terms, Dictionary<string, float> fuzzyOutput)
215 {
216 const int STEPS = 50;
217 float numerator = 0f, denominator = 0f;
218
219 for (int i = 0; i <= STEPS; i++)
220 {
221 float z = (float)i / STEPS;
222 float μz = 0f;
223 // μ(z) = max per ogni term di min(activation, membership(z))
224 foreach (var term in terms)
225 {
226 if (!fuzzyOutput.TryGetValue(term.label, out var α))
227 continue;
228 var μt = FuzzifyTerm(z, term.points);
229 μz = Mathf.Max(μz, Mathf.Min(α, μt));
230 }
231 numerator += μz * z;
232 denominator += μz;
233 }
234 return denominator > 0f ? numerator / denominator : 0f;
235 }
